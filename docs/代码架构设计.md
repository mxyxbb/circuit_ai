### 1.2 MVVM 数据流

```
 ┌─────────┐  command   ┌──────────────┐  control   ┌────────────┐
 │         │───────────>│              │───────────>│            │
 │  View   │            │  ViewModel   │            │  Simulator │
 │ (ImGui) │<───────────│              │<───────────│  (Engine)  │
 │         │  observe   │  ScopeModel  │  data      │            │
 └─────────┘            └──────┬───────┘            └─────┬──────┘
                               │                          │
                     PlotArea → MuxEntry           SPSC RingBuffer
                     → ScrollingBuffer             (引擎线程写入)
                     (UI线程读取)
```

**关键数据通路**:
1. 用户操作 View → 调用 ViewModel 方法（如 `play()`, `pause()`）
2. ViewModel 控制仿真线程启停
3. 仿真线程每步推算 `SimSample` 到 SPSC RingBuffer
4. ViewModel 每帧从 RingBuffer 消费数据，追加到 ScrollingBuffer
5. View 每帧从 ScrollingBuffer 零拷贝读取，ImPlot 渲染

### 3.2 无锁环形缓冲区 — `common/ring_buffer.h`

```cpp
#pragma once
#include <array>
#include <atomic>
#include <cstddef>

// 单生产者单消费者无锁环形缓冲区
// T 必须是可平凡拷贝的类型
template<typename T, size_t Capacity>
class SPSCRingBuffer {
    static_assert((Capacity & (Capacity - 1)) == 0, "Capacity must be power of 2");

public:
    bool push(const T& item) {
        const size_t head = head_.load(std::memory_order_relaxed);
        const size_t tail = tail_.load(std::memory_order_acquire);
        if (head - tail >= Capacity) return false; // full
        buffer_[head & (Capacity - 1)] = item;
        head_.store(head + 1, std::memory_order_release);
        return true;
    }

    bool pop(T& item) {
        const size_t tail = tail_.load(std::memory_order_relaxed);
        const size_t head = head_.load(std::memory_order_acquire);
        if (tail >= head) return false; // empty
        item = buffer_[tail & (Capacity - 1)];
        tail_.store(tail + 1, std::memory_order_release);
        return true;
    }

    bool empty() const {
        return tail_.load(std::memory_order_acquire) >=
               head_.load(std::memory_order_acquire);
    }

private:
    std::array<T, Capacity> buffer_;
    alignas(64) std::atomic<size_t> head_{0}; // cache line隔离避免false sharing
    alignas(64) std::atomic<size_t> tail_{0};
};
```

### 3.3 MNA 求解器 — `engine/mna_solver.h`

```cpp
#pragma once
#include <Eigen/Dense>
#include <cstddef>

class MNASolver {
public:
    // 初始化：n=节点数, m=需要额外电流变量的数量(电压源+电感+变压器)
    void init(size_t n, size_t m);

    // 清空矩阵和向量（每步开始前调用）
    void clear();

    // 盖章接口
    void stampConductance(size_t i, size_t j, double g);  // G矩阵
    void stampB(size_t node, size_t extra_idx, double v);  // B矩阵
    void stampC(size_t extra_idx, size_t node, double v);  // C矩阵
    void stampRhs(size_t idx, double val);                 // 右侧向量b

    // 求解 Ax=b，返回解向量x的const引用
    const Eigen::VectorXd& solve();

    // 获取解向量的特定元素
    double getNodeVoltage(size_t node) const;
    double getExtraCurrent(size_t idx) const;

    size_t matrixSize() const { return n_ + m_; }

private:
    size_t n_ = 0;  // 节点数(不含GND)
    size_t m_ = 0;  // 额外变量数
    Eigen::MatrixXd A_;      // 系统矩阵 (n+m) x (n+m)
    Eigen::VectorXd b_;      // 右侧向量 (n+m)
    Eigen::VectorXd x_;      // 解向量   (n+m)
};
```

**盖章规则说明**：

节点编号约定：0=GND(不进入矩阵)，1~n 为未知节点。
矩阵下标：节点 i 对应行/列 i-1；额外变量 j 对应行/列 n+j。

### 3.4 器件基类 — `components/base_component.h`

```cpp
#pragma once
#include <string>
#include <vector>

class MNASolver;

class BaseComponent {
public:
    virtual ~BaseComponent() = default;

    // 器件标识
    virtual std::string name() const = 0;

    // 器件需要额外申请的 MNA 变量数（电压源/电感/变压器需要）
    virtual size_t extraVariableCount() const { return 0; }

    // 向 MNA 矩阵注入 stamps
    // dt: 当前步长, t: 当前仿真时间, extra_offset: 该器件的额外变量起始索引
    virtual void stamp(MNASolver& solver, double dt, double t, size_t extra_offset) = 0;

    // 非线性器件：根据最新解向量判断状态是否突变
    // 返回 true 表示状态改变，需要重新求解
    virtual bool updateState(const Eigen::VectorXd& x, size_t extra_offset) { return false; }

    // 储能器件：收敛后提交历史状态
    virtual void commitHistory(const Eigen::VectorXd& x, size_t extra_offset) {}

    // 探针数据：该器件可提供的观测电流值
    virtual double getBranchCurrent(const Eigen::VectorXd& x, size_t extra_offset) const { return 0.0; }
};
```

### 3.6 仿真调度器 — `engine/simulator.h`

```cpp
#pragma once
#include "common/sim_types.h"
#include "common/ring_buffer.h"
#include <atomic>
#include <memory>
#include <thread>
#include <vector>

class BaseComponent;
class MNASolver;
class Circuit;

class Simulator {
public:
    Simulator();
    ~Simulator();

    // 从 Circuit 构建仿真环境
    bool setup(const Circuit& circuit, const SimConfig& config,
               const std::vector<SignalInfo>& probes);

    // 线程控制
    void start();      // 启动仿真线程
    void pause();      // 暂停
    void resume();     // 继续
    void reset();      // 重置到 t=0
    void stop();       // 停止线程

    // 状态查询
    bool isRunning() const { return running_.load(); }
    bool isPaused() const  { return paused_.load(); }
    double currentTime() const { return t_.load(); }

    // 数据消费：UI线程调用
    bool consumeSample(SimSample& sample) { return ringBuffer_.pop(sample); }

private:
    void runLoop();    // 仿真线程主循环
    bool step();       // 执行单步，返回是否收敛

    std::unique_ptr<MNASolver> solver_;
    std::vector<std::unique_ptr<BaseComponent>> components_;
    SimConfig config_;
    std::vector<SignalInfo> probes_;

    // 线程同步
    std::thread thread_;
    std::atomic<bool> running_{false};
    std::atomic<bool> paused_{false};
    std::atomic<bool> stopRequested_{false};
    std::atomic<double> t_{0.0};

    // 数据桥
    SPSCRingBuffer<SimSample, 4096> ringBuffer_;

    // 内部状态
    size_t extraVarCount_ = 0;
    size_t stepCount_ = 0;
};
```

**step() 伪代码**：

```
1. solver_->clear()
2. for each component: stamp(solver, dt, t, extra_offset)
3. solver_->solve() → x
4. converged = true
5. for each nonlinear component:
     if updateState(x, extra_offset): converged = false
6. if not converged and iter < MAX_ITER: goto 1
7. for each storage component: commitHistory(x, extra_offset)
8. if (stepCount_ % sampleRatio == 0):
     build SimSample from x, push to ringBuffer_
9. t_ += dt; stepCount_++
```

### 4.2 Scope 数据模型 — `view_model/scope_model.h`

Scope 的核心数据结构，管理多个坐标区及其信号绑定关系。

```cpp
#pragma once
#include "scrolling_buffer.h"
#include <string>
#include <vector>
#include <memory>

// MUX 条目：绑定到某个坐标区的单个信号
struct MuxEntry {
    std::string signalName;   // 如 "V(1)", "I(R1)"
    ImU32 color;              // 绘制颜色
    bool visible = true;
    ScrollingBuffer buffer;   // 该信号的波形数据

    explicit MuxEntry(const std::string& name, ImU32 col,
                      size_t capacity = 10000)
        : signalName(name), color(col), buffer(capacity) {}
};

// 坐标区：一个独立的 ImPlot，包含一组 MUX 信号
struct PlotArea {
    std::vector<std::unique_ptr<MuxEntry>> entries;
    std::string title;          // 如 "Plot 0"
    bool autoFitY = true;

    explicit PlotArea(const std::string& t) : title(t) {}

    // 查找信号
    MuxEntry* findEntry(const std::string& name);
    void addSignal(const std::string& name, ImU32 color);
    void removeSignal(const std::string& name);
};

// Scope：管理所有坐标区的顶层容器
class ScopeModel {
public:
    ScopeModel();

    // 坐标区管理
    int  plotCount() const { return static_cast<int>(plots_.size()); }
    PlotArea* getPlot(int index);
    const PlotArea* getPlot(int index) const;

    // 在指定位置插入坐标区，返回新坐标区索引
    // insertAfterIdx = -1 表示插入到最顶部
    int insertPlot(int insertAfterIdx, const std::string& title = "");
    void removePlot(int index);   // 至少保留一个 Plot
    int  selectedPlot() const { return selectedPlot_; }
    void setSelectedPlot(int idx) { selectedPlot_ = idx; }

    // 信号管理（指定 Plot）
    void addSignalToPlot(int plotIdx, const std::string& name, ImU32 color);
    void removeSignalFromPlot(int plotIdx, const std::string& name);

    // 全局操作
    void clearAllBuffers();
    void autoFitAll();

    // 颜色分配：为信号自动分配颜色
    static ImU32 nextColor();

private:
    std::vector<std::unique_ptr<PlotArea>> plots_;
    int selectedPlot_ = 0;
    static int colorIndex_;  // 颜色轮转索引
};
```

**Scope 数据结构层次**：

```
ScopeModel
├── PlotArea[0]  "Plot 0"        ← selectedPlot_
│   ├── MuxEntry "V(1)"  color=#FF0000  ScrollingBuffer {x[], y[]}
```

## 5. 入口程序 — `app/main.cpp`

```cpp
#include "frontend/views/main_view.h"
#include "frontend/view_model/main_view_model.h"
#include "imgui.h"
#include "imgui_impl_glfw.h"
#include "imgui_impl_opengl3.h"
#include "implot.h"
#include <GLFW/glfw3.h>

int main() {
    // 1. GLFW + OpenGL 初始化
    glfwInit();
    GLFWwindow* window = glfwCreateWindow(1600, 900, "CircuitAI", nullptr, nullptr);
    glfwMakeContextCurrent(window);
    glfwSwapInterval(1); // VSync

    // 2. ImGui + ImPlot 初始化
    IMGUI_CHECKVERSION();
    ImGui::CreateContext();
    ImPlot::CreateContext();
    ImGui_ImplGlfw_InitForOpenGL(window, true);
    ImGui_ImplOpenGL3_Init("#version 130");

    // 3. 创建 ViewModel 和 MainView
    MainViewModel viewModel;
    MainView mainView;

    // 4. 主循环
    while (!glfwWindowShouldClose(window)) {
        glfwPollEvents();

        ImGui_ImplOpenGL3_NewFrame();
        ImGui_ImplGlfw_NewFrame();
        ImGui::NewFrame();

        // MVVM: ViewModel 更新数据 → View 渲染
        viewModel.update();
        mainView.render(viewModel);

        ImGui::Render();
        int displayW, displayH;
        glfwGetFramebufferSize(window, &displayW, &displayH);
        glViewport(0, 0, displayW, displayH);
        glClearColor(0.1f, 0.1f, 0.1f, 1.0f);
        glClear(GL_COLOR_BUFFER_BIT);
        ImGui_ImplOpenGL3_RenderDrawData(ImGui::GetDrawData());
        glfwSwapBuffers(window);
    }

    // 5. 清理
    ImGui_ImplOpenGL3_Shutdown();
    ImGui_ImplGlfw_Shutdown();
    ImPlot::DestroyContext();
    ImGui::DestroyContext();
    glfwDestroyWindow(window);
    glfwTerminate();
    return 0;
}
```
